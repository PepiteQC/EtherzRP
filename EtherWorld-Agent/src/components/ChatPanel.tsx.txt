import { Button, Textarea, Badge, ScrollArea } from '@blinkdotnew/ui'
import {
  Send, Bot, User, Code2, Shield, Info,
  Loader2, CheckCircle2, XCircle, ChevronDown, StopCircle,
  Terminal, AlertTriangle,
} from 'lucide-react'
import { createCodingAgent, createAskAgent } from '../lib/agent'
import { blink } from '../lib/blink'

// ─── Simple Markdown renderer ────────────────────────────────────────────────
// Handles: **bold**, `inline code`, ```code blocks```, and line breaks.
// Replace with react-markdown if you add that dependency.
function MarkdownContent({ content }: { content: string }) {
  const lines = content.split('\n')
  const elements: React.ReactNode[] = []
  let i = 0

  while (i < lines.length) {
    const line = lines[i]

    // Fenced code block
    if (line.startsWith('```')) {
      const lang = line.slice(3).trim()
      const codeLines: string[] = []
      i++
      while (i < lines.length && !lines[i].startsWith('```')) {
        codeLines.push(lines[i])
        i++
      }
      elements.push(
        <pre
          key={i}
          className="bg-muted/80 border border-border rounded-lg p-3 my-2 overflow-x-auto text-xs font-mono leading-relaxed"
        >
          {lang && (
            <div className="text-[10px] text-muted-foreground mb-2 uppercase tracking-wider font-sans">
              {lang}
            </div>
          )}
          <code>{codeLines.join('\n')}</code>
        </pre>
      )
      i++ // skip closing ```
      continue
    }

    // Inline formatting (bold + inline code)
    const formatted = parseInline(line)
    if (line.trim() === '') {
      elements.push(<div key={i} className="h-2" />)
    } else {
      elements.push(
        <p key={i} className="leading-relaxed mb-0.5">
          {formatted}
        </p>
      )
    }
    i++
  }
  return <div className="text-sm space-y-0.5">{elements}</div>
}

function parseInline(text: string): React.ReactNode[] {
  const parts: React.ReactNode[] = []
  // Split on **bold** and `code`
  const regex = /(\*\*(.+?)\*\*|`([^`]+)`)/g
  let last = 0
  let match: RegExpExecArray | null

  while ((match = regex.exec(text)) !== null) {
    if (match.index > last) parts.push(text.slice(last, match.index))
    if (match[0].startsWith('**')) {
      parts.push(<strong key={match.index} className="font-semibold text-foreground">{match[2]}</strong>)
    } else {
      parts.push(
        <code key={match.index} className="px-1.5 py-0.5 bg-muted border border-border rounded text-[11px] font-mono text-accent">
          {match[3]}
        </code>
      )
    }
    last = match.index + match[0].length
  }
  if (last < text.length) parts.push(text.slice(last))
  return parts
}

// ─── Tool call badge ──────────────────────────────────────────────────────────
type ToolCallStatus = 'running' | 'done' | 'error'

interface ToolCallBadgeProps {
  toolName: string
  status: ToolCallStatus
}

function ToolCallBadge({ toolName, status }: ToolCallBadgeProps) {
  const icons: Record<ToolCallStatus, React.ReactNode> = {
    running: <Loader2 className="w-3 h-3 animate-spin text-primary" />,
    done:    <CheckCircle2 className="w-3 h-3 text-green-500" />,
    error:   <XCircle className="w-3 h-3 text-destructive" />,
  }
  const labels: Record<ToolCallStatus, string> = {
    running: `Running ${toolName}…`,
    done:    `${toolName} completed`,
    error:   `${toolName} failed`,
  }

  return (
    <div className="flex items-center gap-2 mt-1.5 px-3 py-1.5 bg-muted/50 border border-border rounded-lg text-[10px] text-muted-foreground w-fit">
      {icons[status]}
      <Terminal className="w-3 h-3" />
      <span>{labels[status]}</span>
    </div>
  )
}

// ─── Props ────────────────────────────────────────────────────────────────────
interface ChatPanelProps {
  sandbox: any
  projectId?: string
  userId?: string
  onFinish: () => void
}

// ─── Component ───────────────────────────────────────────────────────────────
export function ChatPanel({ sandbox, projectId, userId, onFinish }: ChatPanelProps) {
  const [mode, setMode] = useState<'agent' | 'ask'>('agent')
  const [errorMsg, setErrorMsg] = useState<string | null>(null)
  const { isAuthenticated, isLoading: isAuthLoading } = useBlinkAuth()

  const scrollRef    = useRef<HTMLDivElement>(null)
  // Track which message indices have already been persisted to avoid duplicates
  const savedIndices = useRef<Set<number>>(new Set())

  // Agents are scoped to the current project and user so memory is isolated.
  // useMemo ensures we don't recreate them on every render — only when the
  // project or user changes (e.g. switching projects).
  const agents = useMemo(() => {
    const pid = projectId ?? '__default__'
    const uid = userId   ?? '__default__'
    return {
      coding: createCodingAgent(pid, uid),
      ask:    createAskAgent(pid, uid),
    }
  }, [projectId, userId])

  const { data: dbMessages = [] } = useChatMessages(projectId)
  const saveMessage = useSaveMessage()

  const { messages, input, setInput, handleSubmit, isLoading, stop } = useAgent({
    agent: mode === 'agent' ? agents.coding : agents.ask,
    sandbox,
    onFinish: () => {
      setErrorMsg(null)
      onFinish()
    },
    initialMessages: dbMessages.map(m => ({
      role: m.role as any,
      content: m.content,
      toolCalls: m.toolCalls ? JSON.parse(m.toolCalls) : undefined,
    })),
  })

  // ── Persist new messages (de-duped by index) ──────────────────────────────
  useEffect(() => {
    if (!projectId || !userId) return
    messages.forEach((msg, idx) => {
      if (savedIndices.current.has(idx)) return
      // Only save user/assistant messages that have content
      if ((msg.role === 'user' || msg.role === 'assistant') && msg.content) {
        savedIndices.current.add(idx)
        saveMessage.mutate({
          projectId,
          userId,
          role: msg.role as any,
          content: msg.content,
          toolCalls: msg.toolCalls ? JSON.stringify(msg.toolCalls) : undefined,
        })
      }
    })
  }, [messages, projectId, userId])

  // ── Auto-scroll ───────────────────────────────────────────────────────────
  useEffect(() => {
    const el = scrollRef.current
    if (!el) return
    // Only auto-scroll if user is near the bottom (within 120px)
    const nearBottom = el.scrollHeight - el.scrollTop - el.clientHeight < 120
    if (nearBottom || isLoading) {
      el.scrollTop = el.scrollHeight
    }
  }, [messages, isLoading])

  // ── Submit ────────────────────────────────────────────────────────────────
  const onChatSubmit = useCallback(async (e: React.FormEvent | React.KeyboardEvent) => {
    e.preventDefault()
    if (!input.trim() || isLoading) return
    setErrorMsg(null)

    if (isAuthLoading) return
    if (!isAuthenticated) {
      blink.auth.login(window.location.href)
      return
    }

    try {
      await handleSubmit()
    } catch (error: any) {
      const isAuthError =
        error?.details?.originalError?.name === 'BlinkAuthError' ||
        error?.message?.includes('401') ||
        error?.message?.includes('Unauthorized')

      if (isAuthError) {
        blink.auth.login(window.location.href)
        return
      }
      console.error('Chat error:', error)
      setErrorMsg(error?.message ?? 'Something went wrong. Please try again.')
    }
  }, [input, isLoading, isAuthLoading, isAuthenticated, handleSubmit])

  // ── Quick prompts ─────────────────────────────────────────────────────────
  const quickPrompts = [
    { label: '🚀 Build a dashboard', value: 'Build a modern dashboard with charts and stats cards' },
    { label: '🔍 Review my code',    value: 'Analyze the current codebase and suggest improvements' },
    { label: '🗄️ Add a database',   value: 'Set up a database with CRUD endpoints using Hono + Blink' },
    { label: '🎨 Improve the UI',    value: 'Redesign the UI to be more modern and polished' },
  ]

  return (
    <div className="flex flex-col h-full bg-background border-r border-border w-[420px] shrink-0">

      {/* ── Header ────────────────────────────────────────────────────────── */}
      <div className="p-4 border-b border-border flex items-center justify-between bg-secondary/30 shrink-0">
        <div className="flex items-center gap-2">
          <div className="w-8 h-8 rounded-lg bg-primary/20 flex items-center justify-center border border-primary/30">
            <Code2 className="w-5 h-5 text-primary" />
          </div>
          <div>
            <h2 className="text-sm font-semibold">EtherWorld AI</h2>
            <p className="text-[10px] text-muted-foreground uppercase tracking-wider">
              {mode === 'agent' ? 'Full Agent · can write & run code' : 'Ask · read-only analysis'}
            </p>
          </div>
        </div>

        {/* Mode toggle */}
        <div className="flex bg-muted p-1 rounded-lg border border-border">
          {(['agent', 'ask'] as const).map(m => (
            <button
              key={m}
              onClick={() => setMode(m)}
              className={`px-3 py-1 text-[10px] font-medium rounded-md transition-all ${
                mode === m
                  ? 'bg-primary text-primary-foreground shadow-sm'
                  : 'text-muted-foreground hover:text-foreground'
              }`}
            >
              {m === 'agent' ? 'Agent' : 'Ask'}
            </button>
          ))}
        </div>
      </div>

      {/* ── Messages ──────────────────────────────────────────────────────── */}
      <div ref={scrollRef} className="flex-1 overflow-y-auto p-4 space-y-5">

        {/* Empty state */}
        {messages.length === 0 && (
          <div className="h-full flex flex-col items-center justify-center text-center space-y-4 animate-fade-in">
            <div className="w-14 h-14 rounded-2xl bg-primary/10 flex items-center justify-center border border-primary/20">
              <Bot className="w-7 h-7 text-primary" />
            </div>
            <div className="max-w-[280px]">
              <h3 className="text-base font-semibold mb-1">EtherWorld ready</h3>
              <p className="text-xs text-muted-foreground">
                {mode === 'agent'
                  ? 'I can write code, install packages, and deploy — just tell me what to build.'
                  : 'Ask me anything about the codebase. I\'ll read files and give you architectural insights.'}
              </p>
            </div>
            <div className="grid grid-cols-2 gap-2 w-full pt-2">
              {quickPrompts.map(p => (
                <button
                  key={p.value}
                  onClick={() => setInput(p.value)}
                  className="p-3 text-[11px] bg-secondary/50 hover:bg-secondary border border-border rounded-xl text-left transition-colors leading-snug"
                >
                  {p.label}
                </button>
              ))}
            </div>
          </div>
        )}

        {/* Message list */}
        {messages.map((message, i) => {
          const isUser = message.role === 'user'
          // Determine tool call statuses:
          // if isLoading and this is the last message, last tool call is still running
          const toolCalls = message.toolCalls ?? []

          return (
            <div key={i} className={`flex gap-3 animate-slide-up ${isUser ? 'flex-row-reverse' : ''}`}>
              {/* Avatar */}
              <div className={`w-7 h-7 rounded-lg flex items-center justify-center shrink-0 border mt-0.5 ${
                isUser
                  ? 'bg-primary/10 border-primary/20 text-primary'
                  : 'bg-secondary border-border text-muted-foreground'
              }`}>
                {isUser ? <User className="w-3.5 h-3.5" /> : <Bot className="w-3.5 h-3.5" />}
              </div>

              {/* Bubble + tool calls */}
              <div className={`flex flex-col gap-1 max-w-[88%] ${isUser ? 'items-end' : ''}`}>
                <div className={isUser ? 'chat-bubble-user' : 'chat-bubble-ai'}>
                  {message.content
                    ? <MarkdownContent content={message.content} />
                    : !isLoading && (
                        <span className="text-xs text-muted-foreground italic">
                          (no text output)
                        </span>
                      )
                  }
                  {/* Typing indicator for the last AI message while loading */}
                  {!isUser && isLoading && i === messages.length - 1 && !message.content && (
                    <div className="flex gap-1 items-center h-4">
                      <span className="w-1.5 h-1.5 rounded-full bg-primary animate-bounce [animation-delay:0ms]" />
                      <span className="w-1.5 h-1.5 rounded-full bg-primary animate-bounce [animation-delay:150ms]" />
                      <span className="w-1.5 h-1.5 rounded-full bg-primary animate-bounce [animation-delay:300ms]" />
                    </div>
                  )}
                </div>

                {/* Tool calls */}
                {toolCalls.map((call: any, idx: number) => {
                  const isLastMsg   = i === messages.length - 1
                  const isLastCall  = idx === toolCalls.length - 1
                  const status: ToolCallStatus =
                    isLoading && isLastMsg && isLastCall ? 'running' : 'done'
                  return (
                    <ToolCallBadge
                      key={idx}
                      toolName={call.toolName ?? call.name ?? 'tool'}
                      status={status}
                    />
                  )
                })}
              </div>
            </div>
          )
        })}

        {/* Error banner */}
        {errorMsg && (
          <div className="flex items-start gap-2 p-3 bg-destructive/10 border border-destructive/30 rounded-xl text-xs text-destructive animate-slide-up">
            <AlertTriangle className="w-3.5 h-3.5 mt-0.5 shrink-0" />
            <span>{errorMsg}</span>
          </div>
        )}
      </div>

      {/* ── Input area ────────────────────────────────────────────────────── */}
      <div className="p-4 border-t border-border bg-secondary/20 shrink-0">
        <form onSubmit={onChatSubmit} className="relative">
          <div className="p-3 bg-background border border-border rounded-2xl focus-within:border-primary/50 focus-within:ring-4 focus-within:ring-primary/10 transition-all shadow-sm">
            <Textarea
              value={input}
              onChange={e => setInput(e.target.value)}
              placeholder={
                mode === 'agent'
                  ? 'What should I build?'
                  : 'Ask about the architecture, a bug, or the code structure…'
              }
              className="w-full bg-transparent border-none outline-none focus:outline-none focus:ring-0 resize-none min-h-[40px] max-h-[160px] text-sm"
              onKeyDown={e => {
                if (e.key === 'Enter' && !e.shiftKey) {
                  e.preventDefault()
                  onChatSubmit(e)
                }
              }}
            />
            <div className="flex items-center justify-between mt-2 pt-2 border-t border-border/50">
              <div className="flex items-center gap-1.5 text-[10px] text-muted-foreground">
                <Shield className="w-3 h-3 text-accent" />
                <span>Sandboxed execution</span>
                <span className="mx-1">·</span>
                <span>{mode === 'agent' ? '50 steps max' : '20 steps max'}</span>
              </div>

              <div className="flex items-center gap-2">
                {/* Stop button when agent is running */}
                {isLoading && (
                  <Button
                    type="button"
                    variant="ghost"
                    size="sm"
                    onClick={stop}
                    className="h-8 px-2 text-xs text-muted-foreground hover:text-destructive gap-1"
                  >
                    <StopCircle className="w-3.5 h-3.5" />
                    Stop
                  </Button>
                )}
                <Button
                  type="submit"
                  size="sm"
                  disabled={!input.trim() || isLoading}
                  className="h-8 w-8 rounded-xl p-0"
                >
                  {isLoading
                    ? <Loader2 className="w-4 h-4 animate-spin" />
                    : <Send className="w-4 h-4" />
                  }
                </Button>
              </div>
            </div>
          </div>
        </form>
        <p className="text-[10px] text-center text-muted-foreground mt-2">
          <kbd className="px-1 py-0.5 rounded bg-muted border border-border">Enter</kbd> envoyer
          &nbsp;·&nbsp;
          <kbd className="px-1 py-0.5 rounded bg-muted border border-border">Shift+Enter</kbd> nouvelle ligne
        </p>
      </div>
    </div>
  )
}
