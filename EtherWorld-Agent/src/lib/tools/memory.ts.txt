/**
 * lib/tools/memory.ts
 *
 * Persistent memory tools for the EtherWorld agent.
 *
 * The agent uses these to build up a knowledge base about:
 *   - The project structure and architecture decisions
 *   - User preferences (preferred libraries, naming conventions, etc.)
 *   - Mistakes made and how they were resolved
 *   - Completed features and their file locations
 *
 * Memory is scoped to a (projectId, userId) pair so each project
 * has its own isolated context. Memories persist across sessions.
 *
 * Tools exposed:
 *   remember(key, value, category)  — store or overwrite a memory
 *   recall(query)                   — retrieve memories matching a query
 *   forget(key)                     — delete a specific memory
 *   list_memories(category?)        — list all memories (optionally filtered)
 *
 * Storage:
 *   Backed by the /api/memories REST endpoint which reads/writes to the
 *   Blink managed DB. Each memory is a simple key-value record with
 *   a category tag for grouping.
 */

import { tool } from '@blinkdotnew/sdk'
import { z } from 'zod'
import { apiFetch } from '../blink'

// ─── Types ────────────────────────────────────────────────────────────────────

export type MemoryCategory =
  | 'architecture'   // High-level design decisions
  | 'preference'     // User style/library preferences
  | 'error'          // Errors encountered + how they were fixed
  | 'feature'        // Completed features and their locations
  | 'context'        // General project context
  | 'instruction'    // Explicit instructions from the user

export interface Memory {
  id:         string
  projectId:  string
  userId:     string
  key:        string
  value:      string
  category:   MemoryCategory
  createdAt:  string
  updatedAt:  string
}

// ─── Runtime scope injector ───────────────────────────────────────────────────
// The agent's tools need to know which project/user they're operating in.
// We inject these at agent startup via createMemoryTools(projectId, userId).

export function createMemoryTools(projectId: string, userId: string) {

  // ── remember ────────────────────────────────────────────────────────────────
  const remember = tool({
    name: 'remember',

    description: `
Store a piece of information in your long-term project memory.

Use this to record:
  - Architecture decisions: "We use Hono for the API layer."
  - User preferences: "User prefers Tailwind over CSS modules."
  - Solved errors: "Bun crashes on top-level await in this config — use async IIFE."
  - Feature locations: "Auth logic is in src/lib/auth.ts, routes in src/routes/auth.ts."
  - Explicit instructions: "User said: always use zod for validation."

Choose a descriptive, stable key (e.g. "db_orm_choice", "auth_strategy").
If a memory with the same key exists, it will be overwritten.
    `.trim(),

    parameters: z.object({
      key: z
        .string()
        .min(2)
        .max(100)
        .regex(/^[a-z0-9_]+$/, 'Use snake_case keys only')
        .describe('Stable snake_case identifier, e.g. "preferred_ui_library"'),

      value: z
        .string()
        .min(1)
        .max(2000)
        .describe('The information to store. Be specific and complete.'),

      category: z
        .enum(['architecture', 'preference', 'error', 'feature', 'context', 'instruction'])
        .describe('Category for grouping and filtering memories.'),
    }),

    execute: async ({ key, value, category }) => {
      try {
        const memory = await apiFetch<Memory>('/api/memories', {
          method: 'POST',
          body: { projectId, userId, key, value, category },
        })
        return {
          status:   'saved',
          key:      memory.key,
          category: memory.category,
          message:  `Memory "${key}" stored successfully.`,
        }
      } catch (err: any) {
        return { status: 'error', message: err.message }
      }
    },
  })

  // ── recall ───────────────────────────────────────────────────────────────────
  const recall = tool({
    name: 'recall',

    description: `
Search your long-term project memory for relevant information.

Call this:
  - At the start of every session to load project context.
  - Before making architecture decisions (check if one was already made).
  - Before installing a library (check if user has a stated preference).
  - Before writing a new file (check if a similar one already exists).
  - When you encounter an error (check if you've solved it before).

The search is keyword-based across both keys and values.
    `.trim(),

    parameters: z.object({
      query: z
        .string()
        .min(2)
        .describe('Keywords to search for, e.g. "database orm" or "auth strategy"'),

      category: z
        .enum(['architecture', 'preference', 'error', 'feature', 'context', 'instruction'])
        .optional()
        .describe('Optional: filter to a specific category.'),
    }),

    execute: async ({ query, category }) => {
      try {
        const params = new URLSearchParams({
          projectId,
          userId,
          q: query,
          ...(category ? { category } : {}),
        })

        const memories = await apiFetch<Memory[]>(`/api/memories/search?${params}`)

        if (memories.length === 0) {
          return {
            status:   'empty',
            message:  `No memories found for query: "${query}"`,
            memories: [],
          }
        }

        return {
          status:   'found',
          count:    memories.length,
          memories: memories.map(m => ({
            key:      m.key,
            value:    m.value,
            category: m.category,
            updated:  m.updatedAt,
          })),
        }
      } catch (err: any) {
        return { status: 'error', message: err.message, memories: [] }
      }
    },
  })

  // ── list_memories ────────────────────────────────────────────────────────────
  const list_memories = tool({
    name: 'list_memories',

    description: `
List all memories for this project, optionally filtered by category.

Use at the start of a session or when you need a complete picture
of everything you know about the project.
    `.trim(),

    parameters: z.object({
      category: z
        .enum(['architecture', 'preference', 'error', 'feature', 'context', 'instruction'])
        .optional()
        .describe('Optional: filter to one category.'),
    }),

    execute: async ({ category }) => {
      try {
        const params = new URLSearchParams({
          projectId,
          userId,
          ...(category ? { category } : {}),
        })

        const memories = await apiFetch<Memory[]>(`/api/memories?${params}`)

        const grouped = memories.reduce<Record<string, typeof memories>>(
          (acc, m) => {
            ;(acc[m.category] ??= []).push(m)
            return acc
          },
          {},
        )

        return {
          status:  'ok',
          total:   memories.length,
          grouped: Object.fromEntries(
            Object.entries(grouped).map(([cat, mems]) => [
              cat,
              mems.map(m => ({ key: m.key, value: m.value })),
            ]),
          ),
        }
      } catch (err: any) {
        return { status: 'error', message: err.message }
      }
    },
  })

  // ── forget ───────────────────────────────────────────────────────────────────
  const forget = tool({
    name: 'forget',

    description: `
Delete a specific memory by key.

Use when:
  - A decision has been reversed (e.g. you switched from REST to GraphQL).
  - A remembered error no longer applies (dependency was updated).
  - The user explicitly asks you to forget something.
    `.trim(),

    parameters: z.object({
      key: z.string().describe('The exact key of the memory to delete.'),
    }),

    execute: async ({ key }) => {
      try {
        await apiFetch(`/api/memories/${encodeURIComponent(key)}?projectId=${projectId}&userId=${userId}`, {
          method: 'DELETE',
        })
        return { status: 'deleted', key, message: `Memory "${key}" removed.` }
      } catch (err: any) {
        return { status: 'error', message: err.message }
      }
    },
  })

  return { remember, recall, list_memories, forget }
}
