/**
 * lib/tools/think.ts
 *
 * The "think" tool gives the agent an explicit reasoning step before it acts.
 *
 * Why this matters:
 *   Without a think tool, agents jump straight to tool calls. This causes:
 *   - Poorly scoped file writes (overwriting things they shouldn't)
 *   - Missed edge cases (no error handling considered)
 *   - Inconsistent decisions across steps (no working memory)
 *
 *   With a think tool, the agent is instructed to pause, reason out loud,
 *   and only then proceed. This mirrors how senior engineers work:
 *   think first, type second.
 *
 * Implementation:
 *   The tool receives the agent's reasoning as a string and simply returns it.
 *   It has no side effects. Its value is entirely in the *forcing function*:
 *   the system prompt instructs the agent to call it before every action.
 *
 * Reference:
 *   Anthropic research shows that giving models a scratchpad to reason in
 *   before acting significantly improves correctness on multi-step tasks.
 *   https://www.anthropic.com/research/claude-think-tool
 */

import { tool } from '@blinkdotnew/sdk'
import { z } from 'zod'

export const thinkTool = tool({
  name: 'think',

  description: `
Use this tool to reason through a problem before taking any action.

Call it:
  1. Before writing or modifying files — reason about what to change and why.
  2. Before running shell commands — consider side effects and failure modes.
  3. When you receive unexpected output — diagnose before retrying.
  4. When the user's request is ambiguous — clarify your interpretation.

The content of your thinking is NOT shown to the user. It is your private
scratchpad. Be honest and exhaustive here — the quality of your actions
depends directly on the quality of your reasoning.
  `.trim(),

  parameters: z.object({
    reasoning: z
      .string()
      .min(10)
      .describe(
        'Your full internal reasoning. Cover: what you understand, ' +
        'what you plan to do, what could go wrong, and how you will verify success.',
      ),
  }),

  // Pure function — no side effects, just echoes reasoning back so it
  // appears in the tool result and the agent can reference it in the next step.
  execute: async ({ reasoning }) => ({
    thought: reasoning,
    status:  'Reasoning logged. Proceed with your plan.',
  }),
})
