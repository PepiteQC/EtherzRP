
/**
 * lib/blink.ts
 *
 * Singleton Blink client used throughout the app.
 *
 * Exposes:
 *   blink.auth    — login / logout / getToken
 *   blink.db      — typed database helpers (projects, messages)
 *   blink.storage — object storage (file uploads)
 *   blink.realtime — pub/sub channel helper
 *
 * Usage:
 *   import { blink } from './lib/blink'
 *   blink.auth.login(window.location.href)
 */
 
import { createBlink } from '@blinkdotnew/sdk'
 
// ─── Client ───────────────────────────────────────────────────────────────────
// projectId is resolved at runtime: env var → hostname → fallback.
function resolveProjectId(): string {
  const envId = import.meta.env.VITE_BLINK_PROJECT_ID
  if (envId) return envId
 
  if (typeof window !== 'undefined') {
    const match = window.location.hostname.match(/^([^.]+)\.sites\.blink\.new$/)
    if (match) return match[1]
  }
 
  return 'etherworld-code-studio-ip0rboj0'
}
 
export const blink = createBlink({
  projectId:      resolveProjectId(),
  publishableKey: import.meta.env.VITE_BLINK_PUBLISHABLE_KEY,
})
 
// ─── Auth helpers (typed wrappers) ────────────────────────────────────────────
 
/** Redirect to Blink login, then return to `returnTo` (defaults to current URL). */
export function login(returnTo: string = window.location.href): void {
  blink.auth.login(returnTo)
}
 
/** Log the current user out and redirect to the home page. */
export function logout(): void {
  blink.auth.logout()
}
 
/**
 * Retrieve the current session JWT.
 * Throws if the user is not authenticated.
 */
export async function getAuthToken(): Promise<string> {
  const token = await blink.auth.getToken()
  if (!token) throw new Error('Not authenticated — no token available.')
  return token
}

// ─── Request helper ───────────────────────────────────────────────────────────
// A thin fetch wrapper that attaches the Bearer token and handles JSON.
// Use this when you need to call your own Hono API routes.
 
interface FetchOptions extends Omit<RequestInit, 'body'> {
  body?: unknown
}
 
export async function apiFetch<T = unknown>(
  path: string,
  options: FetchOptions = {},
): Promise<T> {
  const token = await getAuthToken()
 
  const { body, headers = {}, ...rest } = options
 
  const response = await fetch(path, {
    ...rest,
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${token}`,
      ...(headers as Record<string, string>),
    },
    body: body !== undefined ? JSON.stringify(body) : undefined,
  })
 
  if (!response.ok) {
    const text = await response.text().catch(() => response.statusText)
    throw new Error(`[${response.status}] ${path} — ${text}`)
  }
 
  // Return null for 204 No Content
  if (response.status === 204) return null as T
 
  return response.json() as Promise<T>
}
