/**
 * lib/sandbox.ts
 *
 * Manages the lifecycle of Blink sandboxes:
 *   createSandbox()          — provision a new sandbox, wait until ready
 *   getPreviewUrl(id)        — return the live preview URL for a sandbox
 *   destroySandbox(id)       — terminate a sandbox (cleanup)
 *   waitForSandboxReady(id)  — poll until the sandbox is in the "ready" state
 *
 * Sandbox statuses: "provisioning" → "ready" | "error" | "stopped"
 */

import { blink, apiFetch } from './blink'

// ─── Types ────────────────────────────────────────────────────────────────────

export type SandboxStatus = 'provisioning' | 'ready' | 'error' | 'stopped'

export interface Sandbox {
  /** Unique sandbox identifier */
  id: string
  /** Human-readable name (auto-generated) */
  name: string
  /** Current lifecycle status */
  status: SandboxStatus
  /** ISO timestamp of creation */
  createdAt: string
  /** ISO timestamp of last activity */
  lastActiveAt: string
}

// ─── Constants ────────────────────────────────────────────────────────────────

/** How long to wait (ms) between status polls when provisioning */
const POLL_INTERVAL_MS = 1_500

/** Maximum time (ms) to wait for a sandbox to become ready */
const PROVISION_TIMEOUT_MS = 90_000

/** How many times to retry the initial create call on transient failures */
const CREATE_MAX_RETRIES = 3

/** Base URL for the Blink sandbox API */
const SANDBOX_API = 'https://api.blink.new/v1/sandboxes'

// ─── Internal helpers ─────────────────────────────────────────────────────────

function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms))
}

/** Fetch a sandbox's current record from the API. */
async function fetchSandbox(id: string): Promise<Sandbox> {
  return apiFetch<Sandbox>(`${SANDBOX_API}/${id}`)
}

/**
 * Poll a sandbox until it reaches "ready" or "error".
 * Resolves with the sandbox when ready, rejects on timeout or error status.
 */
async function waitForSandboxReady(id: string): Promise<Sandbox> {
  const deadline = Date.now() + PROVISION_TIMEOUT_MS

  while (Date.now() < deadline) {
    const sandbox = await fetchSandbox(id)

    if (sandbox.status === 'ready') return sandbox

    if (sandbox.status === 'error') {
      throw new Error(`Sandbox ${id} entered error state during provisioning.`)
    }

    if (sandbox.status === 'stopped') {
      throw new Error(`Sandbox ${id} was stopped before it became ready.`)
    }

    // Still provisioning — wait and try again
    await sleep(POLL_INTERVAL_MS)
  }

  throw new Error(
    `Sandbox ${id} did not become ready within ${PROVISION_TIMEOUT_MS / 1000}s.`,
  )
}

// ─── Public API ───────────────────────────────────────────────────────────────

/**
 * Provision a new sandbox and wait for it to be ready.
 *
 * Retries up to CREATE_MAX_RETRIES times on transient network failures.
 *
 * @returns A fully-provisioned Sandbox object.
 * @throws  If the sandbox errors out or the timeout is exceeded.
 *
 * @example
 *   const sandbox = await createSandbox()
 *   console.log(sandbox.id, sandbox.status) // "abc123", "ready"
 */
export async function createSandbox(): Promise<Sandbox> {
  let lastError: Error | null = null

  for (let attempt = 1; attempt <= CREATE_MAX_RETRIES; attempt++) {
    try {
      // Request a new sandbox (Blink auto-selects a region + runtime)
      const created = await apiFetch<Sandbox>(SANDBOX_API, {
        method: 'POST',
        body: {
          runtime: 'bun',        // use Bun as specified in agent preferences
          template: 'vite-react-ts', // start from a Vite + React + TS template
        },
      })

      // If it's already ready (fast-path for cached sandboxes), return immediately
      if (created.status === 'ready') return created

      // Otherwise poll until it transitions to ready
      return await waitForSandboxReady(created.id)
    } catch (err: unknown) {
      lastError = err instanceof Error ? err : new Error(String(err))

      // Don't retry on definitive errors (auth, schema validation, etc.)
      if (
        lastError.message.includes('401') ||
        lastError.message.includes('403') ||
        lastError.message.includes('error state')
      ) {
        throw lastError
      }

      if (attempt < CREATE_MAX_RETRIES) {
        // Exponential back-off: 1s, 2s, 4s, …
        await sleep(1_000 * 2 ** (attempt - 1))
      }
    }
  }

  throw lastError ?? new Error('Failed to create sandbox after max retries.')
}

/**
 * Return the live HTTPS preview URL for a given sandbox.
 *
 * The URL pattern follows Blink's standard sandbox hostname scheme.
 * The sandbox does NOT need to be in the "ready" state for the URL to be
 * generated — but the preview will only respond once the sandbox is running.
 *
 * @example
 *   const url = getPreviewUrl('abc123')
 *   // → "https://abc123.preview.blink.new"
 */
export function getPreviewUrl(sandboxId: string): string {
  if (!sandboxId) throw new Error('getPreviewUrl: sandboxId is required.')
  return `https://${sandboxId}.preview.blink.new`
}

/**
 * Terminate a sandbox and release its resources.
 * Safe to call on an already-stopped sandbox (no-op on 404).
 */
export async function destroySandbox(sandboxId: string): Promise<void> {
  try {
    await apiFetch(`${SANDBOX_API}/${sandboxId}`, { method: 'DELETE' })
  } catch (err: unknown) {
    // Ignore 404 — sandbox already gone
    if (err instanceof Error && err.message.includes('404')) return
    throw err
  }
}
