import React, { useState, useEffect } from 'react'
import { useBlinkAuth } from '@blinkdotnew/react'
import { ChatPanel } from './components/ChatPanel'
import { PreviewPanel } from './components/PreviewPanel'
import { createSandbox, getPreviewUrl } from './lib/sandbox'
import { blink } from './lib/blink'
import { Code2, Code, Terminal, Zap, Globe, Sparkles, Command, Plus, Folder, Trash2, Settings as SettingsIcon } from 'lucide-react'
import { Button, Badge, Tooltip, TooltipTrigger, TooltipContent, TooltipProvider } from '@blinkdotnew/ui'
import { useProjects, useCreateProject, useDeleteProject, Project } from './hooks/useProjects'

export default function App() {
  const [currentProject, setCurrentProject] = useState<Project | null>(null)
  const [sandbox, setSandbox] = useState<any>(null)
  const [previewUrl, setPreviewUrl] = useState<string | null>(null)
  const [isInitializing, setIsInitializing] = useState(false)
  const [showEditor, setShowEditor] = useState(false)
  const { user, isAuthenticated, isLoading: isAuthLoading } = useBlinkAuth()

  const { data: projects = [] } = useProjects(user?.id)
  const createProject = useCreateProject()
  const deleteProject = useDeleteProject()

  const handleStart = async (existingProject?: Project) => {
    if (isAuthLoading) return

    if (!isAuthenticated) {
      blink.auth.login(window.location.href)
      return
    }

    setIsInitializing(true)
    try {
      let activeProject = existingProject
      
      const newSandbox = await createSandbox()
      setSandbox(newSandbox)

      if (!activeProject) {
        const name = `Project ${projects.length + 1}`
        const project = await createProject.mutateAsync({
          name,
          userId: user!.id,
          sandboxId: newSandbox.id
        })
        activeProject = project
      }

      setCurrentProject(activeProject)
      setShowEditor(true)
    } catch (error: any) {
      const isAuthError =
        error?.details?.originalError?.name === 'BlinkAuthError' ||
        error?.message?.includes('401') ||
        error?.message?.includes('Unauthorized')
      
      if (isAuthError) {
        blink.auth.login(window.location.href)
        return
      }

      console.error('Failed to create sandbox:', error)
    } finally {
      setIsInitializing(false)
    }
  }

  const handleDeleteProject = async (id: string) => {
    if (confirm('Are you sure you want to delete this project?')) {
      await deleteProject.mutateAsync({ id, userId: user!.id })
      if (currentProject?.id === id) {
        setShowEditor(false)
        setCurrentProject(null)
      }
    }
  }

  if (!showEditor) {
    return (
      <div className="min-h-screen bg-background flex flex-col overflow-hidden">
        {/* Navigation */}
        <nav className="h-20 border-b border-border px-8 flex items-center justify-between bg-background/50 backdrop-blur-md sticky top-0 z-50">
          <div className="flex items-center gap-3">
            <div className="w-10 h-10 rounded-xl bg-primary flex items-center justify-center shadow-[0_0_20px_rgba(var(--primary),0.3)]">
              <Code2 className="text-primary-foreground w-6 h-6" />
            </div>
            <span className="text-xl font-bold tracking-tight">EtherWorld</span>
          </div>
          <div className="flex items-center gap-6">
            <div className="hidden md:flex items-center gap-6 text-sm font-medium text-muted-foreground">
              <a href="#" className="hover:text-primary transition-colors">Features</a>
              <a href="#" className="hover:text-primary transition-colors">Docs</a>
              <a href="#" className="hover:text-primary transition-colors">Pricing</a>
            </div>
            <div className="h-6 w-px bg-border hidden md:block" />
            {isAuthenticated ? (
              <div className="flex items-center gap-2">
                <span className="text-sm font-medium truncate max-w-[150px]">{user?.email}</span>
                <Button variant="ghost" size="sm" onClick={() => blink.auth.logout()}>Logout</Button>
              </div>
            ) : (
              <Button onClick={() => blink.auth.login()} size="sm">Sign In</Button>
            )}
            <Button onClick={() => handleStart()} disabled={isInitializing} className="rounded-xl px-6">
              {isInitializing ? 'Initializing...' : 'Launch Editor'}
            </Button>
          </div>
        </nav>

        {/* Hero & Project Section */}
        <main className="flex-1 relative overflow-y-auto">
          <div className="absolute inset-0 bg-[radial-gradient(circle_at_50%_0%,hsl(var(--primary)/0.15),transparent_50%)] pointer-events-none" />
          <div className="max-w-7xl mx-auto px-8 pt-24 pb-32 text-center relative z-10">
            <div className="inline-flex items-center gap-2 px-4 py-2 bg-primary/10 border border-primary/20 rounded-full text-primary text-xs font-semibold mb-8 animate-fade-in">
              <Sparkles className="w-3.5 h-3.5" />
              <span>Free, Intelligent, Powerful</span>
            </div>
            <h1 className="text-6xl md:text-8xl font-black tracking-tighter mb-8 leading-[0.9] animate-slide-up">
              THE FUTURE OF <br />
              <span className="text-transparent bg-clip-text bg-gradient-to-r from-primary to-accent">AI CODING</span>
            </h1>
            <p className="text-lg md:text-xl text-muted-foreground max-w-2xl mx-auto mb-12 animate-slide-up [animation-delay:100ms]">
              EtherWorld Coding Master gives you complete ownership of your development environment. 
              Your own personal AI agent, always ready.
            </p>

            {isAuthenticated && projects.length > 0 && (
              <div className="max-w-4xl mx-auto mb-16 animate-slide-up [animation-delay:150ms]">
                <h2 className="text-2xl font-bold mb-6 text-left flex items-center gap-2">
                  <Folder className="w-6 h-6 text-primary" />
                  Your Master Projects
                </h2>
                <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-4">
                  {projects.map(project => (
                    <div key={project.id} className="group relative bg-secondary/30 border border-border rounded-2xl p-6 text-left hover:border-primary/50 transition-all cursor-pointer" onClick={() => handleStart(project)}>
                      <h3 className="font-bold mb-1 truncate pr-8">{project.name}</h3>
                      <p className="text-xs text-muted-foreground">Created {new Date(project.createdAt!).toLocaleDateString()}</p>
                      <button 
                        onClick={(e) => { e.stopPropagation(); handleDeleteProject(project.id); }}
                        className="absolute top-4 right-4 p-2 text-muted-foreground hover:text-destructive opacity-0 group-hover:opacity-100 transition-all"
                      >
                        <Trash2 className="w-4 h-4" />
                      </button>
                    </div>
                  ))}
                  <button 
                    onClick={() => handleStart()}
                    className="flex flex-col items-center justify-center gap-2 border-2 border-dashed border-border rounded-2xl p-6 hover:border-primary hover:bg-primary/5 transition-all group text-muted-foreground hover:text-primary"
                  >
                    <Plus className="w-8 h-8 group-hover:scale-110 transition-transform" />
                    <span className="font-semibold text-sm">New Project</span>
                  </button>
                </div>
              </div>
            )}

            <div className="flex flex-col sm:flex-row items-center justify-center gap-4 animate-slide-up [animation-delay:200ms]">
              <Button size="lg" onClick={() => handleStart()} disabled={isInitializing} className="h-14 px-8 text-lg rounded-2xl group">
                {isInitializing ? 'Warping to Workspace...' : 'Start New Project'}
                <Zap className="ml-2 w-5 h-5 group-hover:fill-current transition-all" />
              </Button>
            </div>
          </div>
        </main>
      </div>
    )
  }

  return (
    <div className="h-screen flex flex-col overflow-hidden bg-background">
      <TooltipProvider delayDuration={0}>
        {/* Top Bar */}
        <div className="h-12 border-b border-border px-4 flex items-center justify-between bg-background shrink-0">
          <div className="flex items-center gap-4">
            <div className="flex items-center gap-2">
              <Code2 className="text-primary w-5 h-5" />
              <span className="text-sm font-bold tracking-tight">EtherWorld Coding Master</span>
            </div>
            <div className="h-4 w-px bg-border" />
            <div className="flex items-center gap-2">
              <Badge variant="outline" className="text-[10px] font-bold py-0 h-5 px-2 bg-secondary border-border">
                {currentProject?.name}
              </Badge>
            </div>
          </div>
          <div className="flex items-center gap-3">
            <Badge variant="outline" className="text-[10px] bg-primary/5 text-primary border-primary/20">
              Personal Agent Active
            </Badge>
            <div className="h-4 w-px bg-border mx-1" />
            <Button variant="ghost" size="sm" className="h-8 text-xs gap-2" onClick={() => setShowEditor(false)}>
              Back to Dashboard
            </Button>
          </div>
        </div>

        <div className="flex-1 flex overflow-hidden">
          {/* Activity Bar (Side) */}
          <div className="w-12 border-r border-border bg-background flex flex-col items-center py-4 gap-4 shrink-0">
            <Tooltip>
              <TooltipTrigger asChild>
                <div className="w-8 h-8 rounded-lg bg-primary/10 flex items-center justify-center text-primary border border-primary/20 shadow-sm cursor-pointer">
                  <Code2 className="w-4 h-4" />
                </div>
              </TooltipTrigger>
              <TooltipContent side="right">Editor</TooltipContent>
            </Tooltip>
            
            <Tooltip>
              <TooltipTrigger asChild>
                <div className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:bg-secondary hover:text-foreground transition-all cursor-pointer">
                  <Terminal className="w-4 h-4" />
                </div>
              </TooltipTrigger>
              <TooltipContent side="right">Terminal</TooltipContent>
            </Tooltip>

            <Tooltip>
              <TooltipTrigger asChild>
                <div className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:bg-secondary hover:text-foreground transition-all cursor-pointer">
                  <Globe className="w-4 h-4" />
                </div>
              </TooltipTrigger>
              <TooltipContent side="right">Live Preview</TooltipContent>
            </Tooltip>

            <div className="mt-auto space-y-4 flex flex-col items-center">
              <Tooltip>
                <TooltipTrigger asChild>
                  <div className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:bg-secondary hover:text-foreground transition-all cursor-pointer">
                    <Zap className="w-4 h-4" />
                  </div>
                </TooltipTrigger>
                <TooltipContent side="right">Agent Settings</TooltipContent>
              </Tooltip>
              
              <Tooltip>
                <TooltipTrigger asChild>
                  <div className="w-8 h-8 rounded-lg flex items-center justify-center text-muted-foreground hover:bg-secondary hover:text-foreground transition-all cursor-pointer">
                    <SettingsIcon className="w-4 h-4" />
                  </div>
                </TooltipTrigger>
                <TooltipContent side="right">Studio Config</TooltipContent>
              </Tooltip>
            </div>
          </div>

          {/* Main Content Area */}
          <ChatPanel 
            sandbox={sandbox} 
            projectId={currentProject?.id}
            userId={user?.id}
            onFinish={() => {
              if (sandbox) setPreviewUrl(getPreviewUrl(sandbox.id))
            }} 
          />
          <PreviewPanel url={previewUrl} isLoading={false} />
        </div>
      </TooltipProvider>
    </div>
  )
}
